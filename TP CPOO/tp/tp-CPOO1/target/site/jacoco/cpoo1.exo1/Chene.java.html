package cpoo1.exo1;

public class Chene extends Arbre{
    private static double prix = 1000;
    private static double agemin = 10;

    public Chene() {
        super();
        this.prix = 1000;
        this.agemin = 10;
    }

    public Chene(double prix, double agemin) {
        super(prix, agemin);
        this.prix = 1000;
        this.agemin = 10;
    }

    @Override
    public double getPrix() {
        return this.prix;
    }

    public double getAgemin() {
        return this.agemin;
    }

    public void setPrix(double prix) {
        this.prix = prix;
    }

    public void setAgemin(double agemin) {
        this.agemin = agemin;
    }

    @Override
    public double defPrix(){
        return prix * volume;
    }
}
