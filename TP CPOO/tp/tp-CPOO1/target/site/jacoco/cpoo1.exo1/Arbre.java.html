package cpoo1.exo1;

public class Arbre {
    protected double prix;
    protected double age;
    protected double volume;
    protected double annee;

    public Arbre(){
        age = 0;
        volume = 0;
    }

    public Arbre(double age, double volume){
        this.age = age;
        this.volume = volume;
    }

    public double get_age(){
        return this.age;
    }

    public double get_volume(){
        return this.volume;
    }

    public void set_age(int age){
        this.age = age;
    }

    public void set_volume(double volume){
        this.volume = volume;
    }

    public void vieillir(){
        age ++;
    }

    public double getPrix(){
        return this.prix * this.volume;
    }

    public double defPrix(){
        return prix * volume;
    }

    public boolean peutEtreCoupe(){
        if (this.age < this.annee){
            return false;
        }
        return true;
    }
}
